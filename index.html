Connect Google Sheets

Every transaction you add will automatically save to your Google Sheet. Follow these steps once:

  1. Open sheets.google.com and create a new sheet named Money Tracker
  2. In Row 1, add these headers exactly:
    Date | Description | Category | Amount | Type | Payment | Notes | Month
  3. Copy your Sheet ID from the URL:
    https://docs.google.com/spreadsheets/d/SHEET_ID_HERE/edit
  4. Open script.google.com → New Project → Paste the Apps Script code below → Deploy as Web App (Anyone can access)
  5. Copy the Web App URL and paste it in the field below
function doPost(e){try{var d=JSON.parse(e.postData.contents);var ss=SpreadsheetApp.openById(d.sheetId);var ws=ss.getSheetByName('Money Tracker')||ss.getActiveSheet();ws.appendRow([d.date,d.description,d.category,d.amount,d.type,d.payment,d.notes,d.month]);return ContentService.createTextOutput(JSON.stringify({status:'ok'})).setMimeType(ContentService.MimeType.JSON);}catch(err){return ContentService.createTextOutput(JSON.stringify({status:'error',message:err.toString()})).setMimeType(ContentService.MimeType.JSON);}}
